实现代码

class Student:
    school = "第一中学"

    def __init__(self, name, age, score=60):
        self.name = name
        self.age = age
        self.score = score

    def study(self):
        self.score += 5
        print(f"{self.name} 学习后成绩: {self.score}")

    def show_info(self):
        print(f"姓名:{self.name}, 年龄:{self.age}, 成绩:{self.score}, 学校:{Student.school}")

# 实例化对象
student1 = Student("小明", 18, 65)
student2 = Student("小红", 17, 80)

# 展示初始信息
student1.show_info()
student2.show_info()

# 调用方法，更新成绩
student1.study()
student2.study()

# 再次展示信息
student1.show_info()
student2.show_info()

# 建立班级
classroom = [student1, student2]
print("班级成员:", [s.name for s in classroom])
